import random

def get_choices():  #Function
  player_choice = input("Enter a choice (rock, paper, sessiors)=")
  options = ["rock", "paper", "sessiors"]
  computer_choice = random.choice(options)
  choices = {"player": player_choice, "computer": computer_choice}

  return choices
